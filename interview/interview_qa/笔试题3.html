    $left = 0;
    $right = $n - 1;

    while ($left < $right) {

        $arr[$left++] = $arr[$right];
        $arr[$right--] = $temp;

function binary_search($array, $n, $value)
{
    $left = 0;
    $right = $n - 1;

    while ($left <= $right) {
        $mid = intval(($left + $right) / 2);
        if ($value > $array[$mid]) {
            $right = $mid + 1;
        } elseif ($value < $array[$mid]) {
            $left = $mid - 1;
        } else {
            return $mid;
        }
    }

    return -1;
}
